def compute_next_best_product(basket_el):
    """
    paramtre : basket_el = liste des lments du panier du consommateur
    retour : next_pdt, proba = prochain produit  recommander, probabilit d'achat. Ou (0,0) si aucun produit n'est trouv.
    
    Description :  partir du panier d'un utilisateur, renvoie le produit  recommander s'il n'est pas trouv 
    dans la liste des associations de la table associe au modle FP-Growth. 
    Pour ce faire, nous recherchons dans la table des associations le produit  recommander  partir de chaque 
    produit individuel dans le panier du consommateur.
    """
    
    for k in basket_el:  # Pour chaque lment dans le panier du consommateur
        k = {k}
        if len(association[association['basket'] == k].values) != 0:  # Si nous trouvons une association correspondante dans la table FP Growth
            next_pdt = list(association[association['basket'] == k]['next_product'].values[0])[0]  # Nous prenons le produit consquent
            if next_pdt not in basket_el:  # Nous vrifions si le client n'a pas dj achet le produit
                proba = association[association['basket'] == k]['proba'].values[0]  # Trouver la probabilit associe
                return next_pdt, proba
    
    return 0, 0  # Retourne (0,0) si aucun produit n'a t trouv.

def find_next_product(basket):
    """
    Paramtre : basket = dataframe du panier du consommateur
    Retour : list_next_pdt, list_proba = liste des prochains lments  recommander et les probabilits d'achat associes.
    
    Description : Fonction principale qui utilise la fonction ci-dessus. Pour chaque client dans le jeu de donnes, 
    nous recherchons une association correspondante dans la table du modle FP Growth. Si aucune association n'est trouve, 
    nous appelons la fonction compute_next_best_product qui recherche des associations de produits individuels.
    Si aucune association individuelle n'est trouve, la fonction retourne (0,0).
    """
    n = basket.shape[0]
    list_next_pdt = []
    list_proba = []
    for i in range(n):  # Pour chaque client
        el = set(basket['StockCode'][i])  # Panier du client
        if len(association[association['basket'] == el].values) != 0:  # Si nous trouvons une association correspondante dans la table FP Growth
            next_pdt = list(association[association['basket'] == el]['next_product'].values[0])[0]  # Nous prenons le produit consquent
            proba = association[association['basket'] == el]['proba'].values[0]  # Probabilit associe dans la table
            list_next_pdt.append(next_pdt)
            list_proba.append(proba)

        elif len(association[association['basket'] == el].values) == 0:  # Si aucun antcdent correspondant  tout le panier n'est trouv dans la table
            next_pdt, proba = compute_next_best_product(basket['StockCode'][i])  # Appel  la fonction prcdente
            list_next_pdt.append(next_pdt)
            list_proba.append(proba)

    return list_next_pdt, list_proba

# Renommer la colonne 'StockCode' en 'Customer basket' dans le DataFrame basket
basket = basket.rename(columns={'StockCode': 'Customer basket'})

# Crer un DataFrame data_stock contenant les donnes uniques bases sur 'StockCode' dans le DataFrame data
data_stock = data.drop_duplicates(subset="StockCode", inplace=False)

# Initialiser des listes vides pour stocker les informations sur les prix et les descriptions
prices = []
description_list = []

# Parcourir chaque ligne du DataFrame basket
for i in range(basket.shape[0]):
    stockcode = basket['Recommended Product'][i]
    probability = basket['Probability'][i]
    
    # Si le code de stock n'est pas 0 (c'est--dire, s'il y a un produit recommand)
    if stockcode != 0:
        # Obtenir le prix unitaire et la description du produit recommand
        unitprice = data_stock[data_stock['StockCode'] == stockcode]['UnitPrice'].values[0]
        description = data_stock[data_stock['StockCode'] == stockcode]['Description'].values[0]
        
        # Estimer le prix en multipliant le prix unitaire par la probabilit
        estim_price = unitprice * probability
        
        # Ajouter le prix estim et la description  leurs listes respectives
        prices.append(estim_price)
        description_list.append(description)
    else:
        # Si le code de stock est 0, ajouter 0 au prix et 'Null'  la description
        prices.append(0)
        description_list.append('Null')

# Ajouter les listes de prix et de descriptions au DataFrame basket

basket['Product description'] = description_list
